import argparse

parser = argparse.ArgumentParser(description='') #Create an argument parser

parser.add_argument(
                       "Word", #Name of the argument
                       nargs='?', #Between 0 and 1 arguments.
                       type=str, #Type of accepted value (string)
                       help='Word to print') #Description in help


args = parser.parse_args() #Parse and list arguments in a variable.


if args.Word: #If the "Word" argument exists
    print(args.Word) #Print it!
